**דוגמאות ליישום ב-Python:**
```python
def get_results_of_race(speed_p, speed_v, speed_t):
    distance = 43872
    players = {"Петя": speed_p, "Вася": speed_v, "Толя": speed_t}
    times = []
    for name, speed in players.items():
        time = distance / speed
        times.append((time, name))

    times.sort() # מיון לפי סדר עולה של זמן
    for index, (time, name) in enumerate(times):
        print(f"{index + 1}. {name}")

speed_p1 = 10
speed_v1 = 5
speed_t1 = 7

print(f"קלט:")
print(f"{speed_p1}")
print(f"{speed_v1}")

print(f"פלט:")
get_results_of_race(speed_p1, speed_v1, speed_t1) # יציג: 1. Петя\n2. Толя\n3. Вася

speed_p2 = 50
speed_v2 = 25
speed_t2 = 75
print(f"קלט:")
print(f"{speed_p2}")
print(f"{speed_v2}")
```
